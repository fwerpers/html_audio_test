\displaystyle \phi_{m}{\left(x,z \right)}

and

\displaystyle \mathbf{\Psi}_{m} = \left[\begin{matrix}0\\\psi_{m}{\left(x,z \right)}\\0\end{matrix}\right]

The displacement \mathbf{u}_m for medium m will be given by the Helmholtz decomposition equation

\displaystyle \mathbf{u}_m = \nabla \phi_m + \nabla \times \mathbf{\Psi}_m = \left[\begin{matrix}\frac{\partial}{\partial x} \phi_{m}{\left(x,z \right)} - \frac{\partial}{\partial z} \psi_{m}{\left(x,z \right)}\\0\\\frac{\partial}{\partial z} \phi_{m}{\left(x,z \right)} + \frac{\partial}{\partial x} \psi_{m}{\left(x,z \right)}\end{matrix}\right]

The strain tensor \mathbf{e} is defined by the Jacobian sum \frac{1}{2}(\mathbf{J_{u}}_m + \mathbf{J_{u}}_m^T)

The Jacobian is given by

\displaystyle \mathbf{J_{u}}_m = \left[\begin{matrix}\frac{\partial^{2}}{\partial x^{2}} \phi_{m}{\left(x,z \right)} - \frac{\partial^{2}}{\partial z\partial x} \psi_{m}{\left(x,z \right)} & 0 & - \frac{\partial^{2}}{\partial z^{2}} \psi_{m}{\left(x,z \right)} + \frac{\partial^{2}}{\partial z\partial x} \phi_{m}{\left(x,z \right)}\\0 & 0 & 0\\\frac{\partial^{2}}{\partial x^{2}} \psi_{m}{\left(x,z \right)} + \frac{\partial^{2}}{\partial z\partial x} \phi_{m}{\left(x,z \right)} & 0 & \frac{\partial^{2}}{\partial z^{2}} \phi_{m}{\left(x,z \right)} + \frac{\partial^{2}}{\partial z\partial x} \psi_{m}{\left(x,z \right)}\end{matrix}\right]

This gives the strain tensor

\displaystyle \mathbf{e} = \left[\begin{matrix}\frac{\partial^{2}}{\partial x^{2}} \phi_{m}{\left(x,z \right)} - \frac{\partial^{2}}{\partial z\partial x} \psi_{m}{\left(x,z \right)} & 0 & \frac{\frac{\partial^{2}}{\partial x^{2}} \psi_{m}{\left(x,z \right)}}{2} - \frac{\frac{\partial^{2}}{\partial z^{2}} \psi_{m}{\left(x,z \right)}}{2} + \frac{\partial^{2}}{\partial z\partial x} \phi_{m}{\left(x,z \right)}\\0 & 0 & 0\\\frac{\frac{\partial^{2}}{\partial x^{2}} \psi_{m}{\left(x,z \right)}}{2} - \frac{\frac{\partial^{2}}{\partial z^{2}} \psi_{m}{\left(x,z \right)}}{2} + \frac{\partial^{2}}{\partial z\partial x} \phi_{m}{\left(x,z \right)} & 0 & \frac{\partial^{2}}{\partial z^{2}} \phi_{m}{\left(x,z \right)} + \frac{\partial^{2}}{\partial z\partial x} \psi_{m}{\left(x,z \right)}\end{matrix}\right]

We get the stress tensor from 2\mu \mathbf{e} + \lambda\text{tr}[\mathbf{e}]\mathbf{I}

\displaystyle \mathbf{T} = \left[\begin{matrix}\lambda_{m} \left(\frac{\partial^{2}}{\partial x^{2}} \phi_{m}{\left(x,z \right)} + \frac{\partial^{2}}{\partial z^{2}} \phi_{m}{\left(x,z \right)}\right) + 2 \mu_{m} \left(\frac{\partial^{2}}{\partial x^{2}} \phi_{m}{\left(x,z \right)} - \frac{\partial^{2}}{\partial z\partial x} \psi_{m}{\left(x,z \right)}\right) & 0 & 2 \mu_{m} \left(\frac{\frac{\partial^{2}}{\partial x^{2}} \psi_{m}{\left(x,z \right)}}{2} - \frac{\frac{\partial^{2}}{\partial z^{2}} \psi_{m}{\left(x,z \right)}}{2} + \frac{\partial^{2}}{\partial z\partial x} \phi_{m}{\left(x,z \right)}\right)\\0 & \lambda_{m} \left(\frac{\partial^{2}}{\partial x^{2}} \phi_{m}{\left(x,z \right)} + \frac{\partial^{2}}{\partial z^{2}} \phi_{m}{\left(x,z \right)}\right) & 0\\2 \mu_{m} \left(\frac{\frac{\partial^{2}}{\partial x^{2}} \psi_{m}{\left(x,z \right)}}{2} - \frac{\frac{\partial^{2}}{\partial z^{2}} \psi_{m}{\left(x,z \right)}}{2} + \frac{\partial^{2}}{\partial z\partial x} \phi_{m}{\left(x,z \right)}\right) & 0 & \lambda_{m} \left(\frac{\partial^{2}}{\partial x^{2}} \phi_{m}{\left(x,z \right)} + \frac{\partial^{2}}{\partial z^{2}} \phi_{m}{\left(x,z \right)}\right) + 2 \mu_{m} \left(\frac{\partial^{2}}{\partial z^{2}} \phi_{m}{\left(x,z \right)} + \frac{\partial^{2}}{\partial z\partial x} \psi_{m}{\left(x,z \right)}\right)\end{matrix}\right]

Assume we have three layers. In the first layer we have an incident wave and a reflected wave. In the second we have a transmitted and a reflected compressional and shear wave. In the third we have a transmitted wave. The potentials are given by

\displaystyle \phi_{1} = P e^{i K_{z1} \left(z - z_{s}\right)} + Q e^{- i K_{z1} \left(z - z_{b}\right)}

\displaystyle \phi_{2} = A e^{i K_{z2} \left(z - z_{b}\right)} + B e^{- i K_{z2} \left(z - z_{a}\right)}

\displaystyle \psi_{2} = C e^{i L_{z2} \left(z - z_{b}\right)} + D e^{- i L_{z2} \left(z - z_{a}\right)}

\displaystyle \phi_{3} = F e^{i K_{z3} \left(z - z_{a}\right)}

The boundary conditions for continuity of normal stress and strain at the air/solid interface give the following equations

\displaystyle i K_{z1} P e^{i K_{z1} \left(z_{b} - z_{s}\right)} - i K_{z1} Q = i A K_{z2} - i B K_{z2} e^{i K_{z2} z_{a} - i K_{z2} z_{b}} + i C k_{x} + i D k_{x} e^{i L_{z2} z_{a} - i L_{z2} z_{b}}

\displaystyle 2 i k_{x} \mu_{1} \left(i K_{z1} P e^{i K_{z1} \left(z_{b} - z_{s}\right)} - i K_{z1} Q\right) = - 2 A K_{z2} k_{x} \mu_{2} + 2 B K_{z2} k_{x} \mu_{2} e^{i K_{z2} z_{a} - i K_{z2} z_{b}} + C L_{z2}^{2} \mu_{2} - C k_{x}^{2} \mu_{2} + D L_{z2}^{2} \mu_{2} e^{i L_{z2} z_{a} - i L_{z2} z_{b}} - D k_{x}^{2} \mu_{2} e^{i L_{z2} z_{a} - i L_{z2} z_{b}}

\displaystyle \left(P e^{2 i K_{z1} z_{b}} + Q e^{i K_{z1} \left(z_{b} + z_{s}\right)}\right) \left(- K_{z1}^{2} \lambda_{1} - 2 K_{z1}^{2} \mu_{1} - k_{x}^{2} \lambda_{1}\right) e^{- i K_{z1} \left(z_{b} + z_{s}\right)} = \left(- A \left(K_{z2}^{2} \lambda_{2} + 2 K_{z2}^{2} \mu_{2} + k_{x}^{2} \lambda_{2}\right) e^{i \left(2 K_{z2} z_{b} + 2 L_{z2} z_{b}\right)} - B \left(K_{z2}^{2} \lambda_{2} + 2 K_{z2}^{2} \mu_{2} + k_{x}^{2} \lambda_{2}\right) e^{i \left(K_{z2} z_{a} + K_{z2} z_{b} + 2 L_{z2} z_{b}\right)} - 2 C L_{z2} k_{x} \mu_{2} e^{i \left(2 K_{z2} z_{b} + 2 L_{z2} z_{b}\right)} + 2 D L_{z2} k_{x} \mu_{2} e^{i \left(2 K_{z2} z_{b} + L_{z2} z_{a} + L_{z2} z_{b}\right)}\right) e^{- i \left(2 K_{z2} z_{b} + 2 L_{z2} z_{b}\right)}
